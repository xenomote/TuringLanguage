program ->
    | 

group ->
    | ID "=" symbol {"," symbol}

block ->
    | ID ":" LINE TAB statements

condition ->
    | symbol
    | ID
    | condition "or" condition

statement ->
    | "if" condition statements {"or" condition statements} ["else" statements]
    | [write ","] travel (["," transition] | LINE statement)

write ->
    | "write" symbol {symbol} ["backwards"] [NUMBER "times"]

travel ->
    | "go" direction [NUMBER "times"] ["until" condition]

transition ->
    | "do" ID
